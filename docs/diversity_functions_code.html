Functions for diversity estimates

2021-04-17

This page includes code implementation of all the functions that are used to calculate different diversity estimates.

Shannon index 

shannon
function (cover_data, remove_column = NULL) 
{
    if (is.null(remove_column)) {
        data <- cover_data[, 1:ncol(cover_data)]
    }
    else {
        data <- cover_data[, (remove_column + 1):ncol(cover_data)]
    }
    total <- rowSums(data)
    shannon_index <- rowSums(-data[, 1:ncol(data)]/total * log(data[, 
        1:ncol(data)]/total), na.rm = TRUE)
    return(shannon_index)
}

When calculating the shannon indexes the “na.rm = TRUE” is necessary to take care of the cells where the cover data is 0. The reason for this is that we use the logarithmic function and the log(0) is not defined, which is why R will give a NA when it is asked to calculate log(0). The “na.rm = TRUE” makes sure that when we sum the rows we don’t take into account the NAs produced by log(0).

Simpson’s index 

simpson
function (cover_data, remove_column = NULL) 
{
    if (is.null(remove_column)) {
        data <- cover_data[, 1:ncol(cover_data)]
    }
    else {
        data <- cover_data[, (remove_column + 1):ncol(cover_data)]
    }
    total <- rowSums(data)
    simpson_index <- rowSums((data[, 1:ncol(data)]/total)^2)
    simpson_index[is.nan(simpson_index)] <- 0
    return(simpson_index)
}

Species richness 

species_richness
function (presence_absence_data, remove_column = NULL) 
{
    if (is.null(remove_column)) {
        species_richness <- rowSums(presence_absence_data)
    }
    else {
        species_richness <- rowSums(presence_absence_data[, (remove_column + 
            1):ncol(presence_absence_data)])
    }
    return(species_richness)
}

Hill Shannon diversity 

hill_shannon
function (cover_data, remove_column = NULL) 
{
    if (is.null(remove_column)) {
        vector <- shannon(cover_data)
        return(exp(vector))
    }
    else {
        vector <- shannon(cover_data, remove_column = remove_column)
        return(exp(vector))
    }
}

Hill Simpson diversity 

hill_simpson
function (cover_data, remove_column = NULL) 
{
    if (is.null(remove_column)) {
        vector <- simpson(cover_data)
        v <- 1/vector
        v[is.infinite(v)] <- 0
        return(v)
    }
    else {
        vector <- simpson(cover_data, remove_column = remove_column)
        a <- 1/vector
        a[is.infinite(a)] <- 0
        return(a)
    }
}
